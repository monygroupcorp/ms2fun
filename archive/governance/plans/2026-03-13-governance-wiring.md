### Gnosis Safe Deployment (Mainnet Fork)
Since we fork mainnet, Safe's canonical contracts are already deployed:

- **SafeSingleton:** `0xd9Db270c1B5E3Bd161E8c8503c55cEABeE709552` (v1.3.0)

Deploy a 1-of-1 Safe owned by deployer, then enable GrandCentral as a module.

### Proposal Flow (Sole Owner)
Since deployer has all shares and `sponsorThreshold` is low:
1. `submitProposal(targets, values, calldatas, expiration, details)` — auto-sponsors if caller has enough shares
2. `submitVote(proposalId, true)` — vote yes
3. `evm_increaseTime` past votingPeriod + gracePeriod
4. `processProposal(proposalId, targets, values, calldatas)` — executes via Safe

### ABI Loading
Frontend `abiLoader.js` loads from `/contracts/out/ContractName.sol/ContractName.json` in local dev. As long as `forge build` has run, ABIs for GrandCentral, ShareOffering, RevenueConductor, StipendConductor, OTCShareEscrow are automatically available.

This file and any imports of it should be removed. The governance routes already use `GrandCentralAdapter` via `serviceFactory.getGrandCentralAdapter()`.

```javascript
  const safeSingletonAbi = [
    'function setup(address[] calldata _owners, uint256 _threshold, address to, bytes calldata data, address fallbackHandler, address paymentToken, uint256 payment, address payable paymentReceiver)',

    'function isModuleEnabled(address module) view returns (bool)',

  const safeSetupIface = new ethers.utils.Interface(safeSingletonAbi)

  // Encode Safe.setup() call: 1-of-1, deployer as sole owner, no delegate call

  const safeContract = new ethers.Contract(safeAddress, safeSingletonAbi, deployer)

  // --- Deploy GrandCentral ---
  const grandCentralAbi = await loadAbi('GrandCentral')
  const grandCentralBytecode = await loadBytecode('GrandCentral')
  const grandCentralFactory = new ethers.ContractFactory(grandCentralAbi, grandCentralBytecode, deployer)

  const INITIAL_SHARES = ethers.utils.parseEther('1000')     // 1000 shares to founder

  const GRACE_PERIOD = 120                                     // 2 minutes (local dev)

    INITIAL_SHARES,

  const ownerSig = ethers.utils.hexConcat([

    ethers.utils.hexZeroPad('0x00', 32),            // s = 0
    '0x01',                                          // v = 1 (pre-approved)

    safeAddress,                    // to: the Safe itself
    0,                              // value
    enableModuleData,               // data: enableModule(grandCentral)
    0,                              // operation: Call
    0,                              // safeTxGas
    0,                              // baseGas
    0,                              // gasPrice
    ethers.constants.AddressZero,   // gasToken
    ethers.constants.AddressZero,   // refundReceiver

  const isEnabled = await safeContract.isModuleEnabled(grandCentral.address)
```

```javascript
// ─────────────────────────────────────────────────────────────────────────────
// Governance helpers
// ─────────────────────────────────────────────────────────────────────────────

/**
 * Submit a proposal, vote yes, advance time, and process it.
 * For single-owner (dictator mode) governance bootstrapping.
 *
 * @param {Object} params
 * @param {Object} params.grandCentral - GrandCentral ethers Contract instance
 * @param {Object} params.provider - ethers JsonRpcProvider
 * @param {address[]} params.targets - Call targets
 * @param {uint256[]} params.values - ETH values for each call
 * @param {bytes[]} params.calldatas - Encoded function calls
 * @param {string} params.details - Human-readable proposal description
 * @param {number} params.votingPeriod - Voting period in seconds (to advance time)
 * @param {number} params.gracePeriod - Grace period in seconds (to advance time)
 * @returns {Promise<number>} Proposal ID
 */
export async function submitAndProcessProposal({
  grandCentral,
  provider,
  targets,
  values,
  calldatas,
  details,
  votingPeriod,
  gracePeriod,
}) {
  // Submit (auto-sponsors if caller has enough shares)
  const submitTx = await grandCentral.submitProposal(
    targets, values, calldatas,
    0, // expiration (0 = no expiration)
    details
  )
  const submitReceipt = await submitTx.wait()
  const submitEvent = submitReceipt.events?.find(e => e.event === 'ProposalSubmitted')
  const proposalId = submitEvent.args.proposalId.toNumber()

  // Vote yes
  await (await grandCentral.submitVote(proposalId, true)).wait()

  // Advance time past voting + grace period
  await provider.send('evm_increaseTime', [votingPeriod + gracePeriod + 1])
  await provider.send('evm_mine', [])

  // Process (execute via Safe)
  await (await grandCentral.processProposal(proposalId, targets, values, calldatas)).wait()

  return proposalId
}

/**
 * Create a proposal to register a factory in MasterRegistry.
 *
 * @param {Object} params
 * @param {Object} params.grandCentral - GrandCentral contract
 * @param {Object} params.provider - ethers provider
 * @param {string} params.registryAddress - MasterRegistry address
 * @param {Array} params.registryAbi - MasterRegistry ABI
 * @param {string} params.factoryAddress - Factory to register
 * @param {string} params.contractType - e.g. 'ERC404'
 * @param {string} params.title - Registry title
 * @param {string} params.displayTitle - Display title
 * @param {string} params.metadataURI - Metadata URI
 * @param {string[]} params.features - Feature list
 * @param {number} params.votingPeriod
 * @param {number} params.gracePeriod
 */
export async function registerFactoryViaProposal({
  grandCentral, provider, registryAddress, registryAbi,
  factoryAddress, contractType, title, displayTitle, metadataURI, features,
  votingPeriod, gracePeriod,
}) {
  const registryIface = new ethers.Interface
    ? new ethers.utils.Interface(registryAbi)
    : new ethers.utils.Interface(registryAbi)
  const calldata = registryIface.encodeFunctionData('registerFactory', [
    factoryAddress, contractType, title, displayTitle, metadataURI, features
  ])

  const proposalId = await submitAndProcessProposal({
    grandCentral, provider,
    targets: [registryAddress],
    values: [0],
    calldatas: [calldata],
    details: `Register ${displayTitle} factory (${factoryAddress})`,
    votingPeriod, gracePeriod,
  })

  console.log(`   Proposal #${proposalId}: Registered ${displayTitle} factory`)
  return proposalId
}

/**
 * Create a proposal to register a vault in MasterRegistry.
 */
export async function registerVaultViaProposal({
  grandCentral, provider, registryAddress, registryAbi,
  vaultAddress, creator, name, metadataURI, targetId,
  votingPeriod, gracePeriod,
}) {
  const registryIface = new ethers.utils.Interface(registryAbi)
  const calldata = registryIface.encodeFunctionData('registerVault', [
    vaultAddress, creator, name, metadataURI, targetId
  ])

  const proposalId = await submitAndProcessProposal({
    grandCentral, provider,
    targets: [registryAddress],
    values: [0],
    calldatas: [calldata],
    details: `Register vault: ${name} (${vaultAddress})`,
    votingPeriod, gracePeriod,
  })

  console.log(`   Proposal #${proposalId}: Registered vault ${name}`)
  return proposalId
}
```

## Task 4: Restructure Deploy to Use Governance for Registration

**Files:**
- Modify: `scripts/local-chain/deploy-contracts.mjs`

This is the critical change: factories and vaults should be registered via GrandCentral proposals instead of direct `registry.registerFactory()` calls.

**Step 1: Move factory/vault registration from deploy to seed**

Currently the deploy script directly calls `registry.registerFactory(...)` and `registry.registerVault(...)`. Since GrandCentral executes through the Safe, the Safe needs to be the MasterRegistry owner (or have permission).

Two approaches:
- **Option A:** Transfer MasterRegistry ownership to the Safe before governance registration. The deploy script deploys everything but doesn't register. The seed script registers via proposals.
- **Option B:** Keep direct registration in deploy (since deployer is the owner), and have governance proposals for future additions only. This is more realistic — you deployed infrastructure first, then transferred control to the DAO.

**Go with Option B** — it matches the real launch flow (deploy as dictator, then DAO takes over). The seed script will:
1. Transfer MasterRegistry ownership to the Safe
2. Use proposals for any additional registrations (or just demonstrate the proposal flow)

The existing factory/vault registrations in deploy stay as-is (deployer doing initial setup before DAO exists).

**Step 2: After GrandCentral deployment, transfer ownership**

Add to the end of Phase 5 in `deploy-contracts.mjs`:

```javascript
    })
    console.log(`   Proposal #${configProposalId}: Governance config confirmed`)

    // --- Phase C: Create Share Offering Tranche ---
    console.log('\n   Phase C: Share Offering...')

    // Create a tranche via ShareOffering conductor
    // The conductor needs to be called by an authorized entity
    // For local dev, we use a proposal to create the tranche

    // ShareOffering.createTranche(pricePerShare, totalShares, startTime, endTime,
    //                              finalizeDeadline, minShares, maxSharesPerAddress, whitelistRoot)
    const block = await provider.getBlock('latest')
    const now = block.timestamp
    const trancheStart = now + 10
    const trancheEnd = now + 86400     // 24 hours
    const finalizeDeadline = now + 172800  // 48 hours

    const shareOfferingIface = new ethers.utils.Interface(shareOfferingAbi)
    const createTrancheCalldata = shareOfferingIface.encodeFunctionData('createTranche', [
      ethers.utils.parseEther('0.01'),    // pricePerShare: 0.01 ETH
      ethers.utils.parseEther('500'),     // totalShares: 500 available
      trancheStart,
      trancheEnd,
      finalizeDeadline,
      ethers.utils.parseEther('1'),       // minShares: 1
      ethers.utils.parseEther('100'),     // maxSharesPerAddress: 100
      ethers.constants.HashZero,          // whitelistRoot (open to all)
    ])

    const trancheProposalId = await submitAndProcessProposal({
      grandCentral, provider,
      targets: [governance.shareOffering],
      values: [0],
      calldatas: [createTrancheCalldata],
      details: 'Create initial share offering tranche (500 shares @ 0.01 ETH)',
      votingPeriod: vp, gracePeriod: gp,
    })
    console.log(`   Proposal #${trancheProposalId}: Share offering tranche created`)

    // Advance time so tranche is active
    await provider.send('evm_increaseTime', [11])
    await provider.send('evm_mine', [])

    // Have a test account buy shares
    const traderSigner = provider.getSigner(TEST_ACCOUNTS.trader.address)
    const shareOfferingAsTrader = new ethers.Contract(governance.shareOffering, shareOfferingAbi, traderSigner)
    const sharesToBuy = 10
    const cost = ethers.utils.parseEther('0.1') // 10 shares * 0.01 ETH

    try {
      await (await shareOfferingAsTrader.commit(
        1, // trancheId (first tranche)
        sharesToBuy,
        [], // no proof (open whitelist)
        { value: cost }
      )).wait()
      console.log(`   Trader committed to buy ${sharesToBuy} shares`)
    } catch (err) {
      console.log(`   ⚠ Share commitment failed: ${err.message.slice(0, 80)}`)
    }

    // --- Phase D: One Active (Unprocessed) Proposal ---
    console.log('\n   Phase D: Create pending proposal for UI...')

    // Submit a proposal but DON'T process it — so the UI has an active proposal to display
    const pendingDetails = 'Proposal to increase voting period to 1 hour (pending)'
    const pendingCalldata = grandCentral.interface.encodeFunctionData('setGovernanceConfig', [
      3600, // 1 hour voting
      gp, 1, ethers.utils.parseEther('1'), 50
    ])

    const pendingTx = await grandCentral.submitProposal(
      [governance.grandCentral], [0], [pendingCalldata],
      0, pendingDetails
    )
    const pendingReceipt = await pendingTx.wait()
    const pendingEvent = pendingReceipt.events?.find(e => e.event === 'ProposalSubmitted')
    const pendingId = pendingEvent.args.proposalId.toNumber()

    // Vote yes but don't advance time or process
    await (await grandCentral.submitVote(pendingId, true)).wait()
    console.log(`   Proposal #${pendingId}: Active (voting in progress)`)

    console.log('\n   ✅ Governance seeding complete!')
    console.log(`      - ${configProposalId + 1} processed proposals`)
    console.log(`      - 1 active proposal (#${pendingId})`)
    console.log(`      - Share offering tranche active`)
    console.log(`      - Treasury funded and swept`)
  }
```
